-- Trainers table
CREATE TABLE IF NOT EXISTS Trainers (
    trainer_id SERIAL PRIMARY KEY,
    trainer_name CHAR NOT NULL
);

-- Routines table
CREATE TABLE IF NOT EXISTS Routines (
    routine_id SERIAL PRIMARY KEY,
    routine_name CHAR NOT NULL
);

-- Schedule table
CREATE TABLE IF NOT EXISTS Schedule (
    schedule_id SERIAL PRIMARY KEY,
    trainer_id INTEGER REFERENCES Trainers(trainer_id),
    member_id INTEGER REFERENCES Members(member_id),
    room_id INTEGER REFERENCES Rooms(room_id),
    date DATE NOT NULL,
    time TIME NOT NULL,
    capacity INTEGER NOT NULL,
    CONSTRAINT fk_schedule_trainer FOREIGN KEY (trainer_id) REFERENCES Trainers(trainer_id),
    CONSTRAINT fk_schedule_member FOREIGN KEY (member_id) REFERENCES Members(member_id),
    CONSTRAINT fk_schedule_room FOREIGN KEY (room_id) REFERENCES Rooms(room_id)
);

-- Rooms table
CREATE TABLE IF NOT EXISTS Rooms (
    room_id SERIAL PRIMARY KEY,
    room_name CHAR NOT NULL,
    booking_id INTEGER REFERENCES Bookings(booking_id)
);

-- Bookings table
CREATE TABLE IF NOT EXISTS Bookings (
    booking_id SERIAL PRIMARY KEY,
    booking_name CHAR,
    member_id INTEGER REFERENCES Members(member_id),
    trainer_id INTEGER REFERENCES Trainers(trainer_id)
);

-- Equipments table
CREATE TABLE IF NOT EXISTS Equipments (
    equipment_id SERIAL PRIMARY KEY,
    equipment_name CHAR NOT NULL,
    equipment_status CHAR NOT NULL
);

-- Admin Staff table
CREATE TABLE IF NOT EXISTS AdminStaff (
    staff_id SERIAL PRIMARY KEY,
    staff_name CHAR NOT NULL
);

-- Members table
CREATE TABLE IF NOT EXISTS Members (
    member_id SERIAL PRIMARY KEY,
    member_name CHAR NOT NULL,
    age INTEGER,
    membership_status CHAR NOT NULL,
    height INTEGER,
    weight INTEGER,
    fitness_goals CHAR,
    goal_weight INTEGER,
    goal_runtime INTEGER,
    max_deadlift INTEGER,
    goal_deadlift INTEGER,
    max_benchPress INTEGER,
    goal_benchPress INTEGER
);
